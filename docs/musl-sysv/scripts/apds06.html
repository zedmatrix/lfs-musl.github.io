D.6. /etc/rc.d/init.d/gen-ifconfig 

#/bin/bash
### BEGIN INIT INFO
# Provides:            gen-ifconfig
# Required-Start:      $local_fs $network
# Required-Stop:       $network
# Default-Start:       S
# Default-Stop:
# Short-Description:   generates dyanmic ifconfig
# Description:         generates dynamic ifconfig on load from iface
### END INIT INFO

. /lib/lsb/init-functions

CONFIG_DIR="/etc/sysconfig"

log_info_msg "Generating dyanmic interface config..."

for iface in $(ls /sys/class/net); do
    [ "$iface" = "lo" ] && continue
    FILE="$CONFIG_DIR/ifconfig.$iface"
    if [ ! -f "$FILE" ]; then
        log_success_msg "Creating $FILE"
        cat > "$FILE" <<EOF
ONBOOT="yes"
IFACE="$iface"
SERVICE="dhcpcd"
DHCP_START="-b -q -h $HOSTNAME"
DHCP_STOP="-k"
EOF
    fi
done

exit 0
# End of gen-ifconfig
